Type
Promise
Example
function getA() {
    return new Promise((resolve, reject) => {
        get('/user/a', (status, res) => {
            if (status == 200) {
                resolve(res)
            } else {
                reject('error')
            }
        })
    })
}

function getB(dataA) {
    return new Promise((resolve, reject) => {
        get('/user/b', dataA, (status, res) => {
            if (status == 200) {
                resolve(res)
            } else {
                reject('error')
            }
        })

    })
}

function getC(dataB) {
    return new Promise((resolve, reject) => {
        get('/user/c', dataB, (status, res) => {
            if (status == 200) {
                resolve(res)
            } else {
                reject('error')
            }
        })

    })
}

promiseOrder([getA,getB,getC]).then((res)=>{
    console.log('success');
    console.log(res);
})
